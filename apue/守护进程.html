                                                                                              守护进程

                                                                                              rz大约 2 分钟

                                                                                              # 守护进程

                                                                                              #include "apue.h"
#include <syslog.h>
#include <fcntl.h>
#include <sys/resource.h>

void daemonize(const char *cmd)
{
	int i, fd0, fd1, fd2;
	pid_t pid;
	struct rlimit rl;
	struct sigaction sa;

	/*
	 * Clear file creation mask.
	 * 清除mask
	 */
	umask(0);

	/*
	 * Get maximum number of file descriptors.
	 * 获得最大fd
	 */
	if (getrlimit(RLIMIT_NOFILE, &rl) < 0)
		err_quit("%s: can't get file limit", cmd);

	/*
	 * Become a session leader to lose controlling TTY.
	 * 父进程关闭，子进程setsid
	 * 子进程继承了父进程的进程组 ID，但获得了一个新的进程 ID，这就保证了子进程不是一个 进程组的组长进程，setsid 调用的先决条件
	 *
	 */
	if ((pid = fork()) < 0)
		err_quit("%s: can't fork", cmd);
	else if (pid != 0) /* parent */
		exit(0);

	/*
	 * 该进程不能是进程组组长进程
	 * setsid后发生 a.新会话的首进程，b.成为一个新进程组的组长进程，c.没有控制终端。(如果在调用setsid前，该进程有控制终端，那么与该终端的联系被解除)
	 */
	setsid();

	/*
	 * Ensure future opens won't allocate controlling TTYs.
	 * 忽略hup挂起信号
	 */
	sa.sa_handler = SIG_IGN;
	sigemptyset(&sa.sa_mask);
	sa.sa_flags = 0;
	if (sigaction(SIGHUP, &sa, NULL) < 0)
		err_quit("%s: can't ignore SIGHUP", cmd);

	//进程已经成为无终端的会话首进程,但它可以重新申请打开一个控制终端（因为会话首进程能与控制终端建立连接）
	//所以fork子进程，终止父进程，这样子进程不是会话首进程，就不能与终端建立连接，来获得控制终端
	if ((pid = fork()) < 0)
		err_quit("%s: can't fork", cmd);
	else if (pid != 0) /* parent 终止父进程 */
		exit(0);

	/*
	 * Change the current working directory to the root so
	 * we won't prevent file systems from being unmounted.
	 * 更改工作目录，如果父进程的工作目录可卸载那就出问题了
	 */
	if (chdir("/") < 0)
		err_quit("%s: can't change directory to /", cmd);

	/*
	 * Close all open file descriptors.
	 * 关闭所以fd
	 */
	if (rl.rlim_max == RLIM_INFINITY)
		rl.rlim_max = 1024;
	for (i = 0; i < rl.rlim_max; i++)
		close(i);

	/*
	 * Attach file descriptors 0, 1, and 2 to /dev/null.
	 * 将fd 0，1，2 指向/dev/null
	 */
	fd0 = open("/dev/null", O_RDWR);//fd 应该是0
	fd1 = dup(0);//fd1 应该是1
	fd2 = dup(0); //fd2 应该是2

	/*
	 * Initialize the log file.
	 *
	 */
	openlog(cmd, LOG_CONS, LOG_DAEMON);
	if (fd0 != 0 || fd1 != 1 || fd2 != 2)
	{
		syslog(LOG_ERR, "unexpected file descriptors %d %d %d",
			   fd0, fd1, fd2);
		exit(1);
	}
	printf("end");
}
int main(int argc, char const *argv[])
{
	daemonize("ll");
	return 0;
}
